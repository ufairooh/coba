   public void autodetail()
    {
        try
        {
            String sql = "select top 1 IDDetTrans FROM DetailTransac order by IDDetTrans desc";
            Statement state = con.createStatement();
            ResultSet rs = state.executeQuery(sql);
            
            if(rs.next())
            {
                String ID = rs.getString("IDDetTrans");
                String potong = ID.substring(1);
                int nextid = Integer.parseInt(potong) + 1;
                
                if (nextid<10)
                {
                    ID = "4000" + nextid;
                }
                else if (nextid < 100)
                {
                    ID = "400" + nextid;
                }
                else if (nextid < 1000)
                {
                    ID = "40" + nextid;
                }
                else if (nextid < 10000)
                {
                    ID = "4" + nextid;
                }
                idDT.setText(ID);
                
            }
            else
            {
                idDT.setText("40001");
            }
         }
        catch (Exception e)
        {
            JOptionPane.showMessageDialog(null,e);
        }
    }
