    public void autoidDT()
    {
        try
        {
            String sql = "select top 1 IDTrans FROM Transac order by IDTrans desc";
            Statement state = con.createStatement();
            ResultSet rs = state.executeQuery(sql);
            
            if(rs.next())
            {
                String ID = rs.getString("IDTrans");
                String potong = ID.substring(1);
                int nextid = Integer.parseInt(potong) + 1;
                
                if (nextid<10)
                {
                    ID = "2000" + nextid;
                }
                else if (nextid < 100)
                {
                    ID = "200" + nextid;
                }
                else if (nextid < 1000)
                {
                    ID = "20" + nextid;
                }
                else if (nextid < 10000)
                {
                    ID = "2" + nextid;
                }
                txtID.setText(ID);
                
            }
            else
            {
                txtID.setText("20001");
            }
         }
        catch (Exception e)
        {
            JOptionPane.showMessageDialog(null,e);
        }
    }
